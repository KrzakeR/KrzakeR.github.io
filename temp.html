C++ - ćwiczenia z wykorzystaniem pętli | 1 |
Napisz program, który pobiera liczby wprowadzane przez użytkownika, aż do momentu, gdy ich suma przekroczy wartość 75. Przed końcem program powinien wyświetlić sumę wprowadzonych liczb. 
#include 
using namespace std;

int wynik=0, kolejna=0;

int main() 
{
    while(wynik<=75)
    {
        cout << "Podaj liczbe : ";
        cin >> kolejna;
        wynik+=kolejna;
    }
    cout << wynik;
    return 0;
}
C++ - ćwiczenia z wykorzystaniem pętli | 2 |
Napisz program, który wypisuje kolejne liczby naturalne oddzielone spacjami z dodatniego przedziału określonego przez użytkownika. 
#include 
using namespace std;

int n, m;

int main() 
{
    cout << "Podaj wartosc minimalna : ";
    cin >> n;
    cout << "Podaj wartosc maksymalna : ";
    cin >> m;

    for (n>0; n<=m; n++)
        cout << n << " ";
    
    return 0;
}
C++ - ćwiczenia z wykorzystaniem pętli | 3 |
Napisz program, który wyświetla kwadraty kolejnych liczb naturalnych, począwszy od zera a skończywszy na kwadracie liczby podanej przez użytkownika. 
#include 
using namespace std;

int n, m=0;

int main() 
{
    cout << "Podaj liczbe : ";
    cin >> n;
        
    for (m=0; m<=n; m++)
        cout << m*m << endl;

    return 0;
}

Przygotuj schemat blokowy algorytmu wypisującego wszystkie dwucyfrowe liczby parzyste podzielne przez 3. 
#include 
using namespace std;

int main() 
{
    for (int i=10; i<=99; i++)
    {
        if (i%3==0)
        cout << i << " ";
    }
    
    return 0;
}
