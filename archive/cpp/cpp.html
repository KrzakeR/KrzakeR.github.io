C++ - ćwiczenia z wykorzystaniem pętli | 1 |
Napisz program, który pobiera liczby wprowadzane przez użytkownika, aż do momentu, gdy ich suma przekroczy wartość 75. Przed końcem program powinien wyświetlić sumę wprowadzonych liczb. 
#include <iostream>
using namespace std;

int wynik=0, kolejna=0;

int main() 
{
    while (wynik<=75)
    {
        cout << "Podaj liczbe : ";
        cin >> kolejna;
        wynik+=kolejna;
    }
    cout << wynik;
    return 0;
}

Napisz program, który wypisuje kolejne liczby naturalne oddzielone spacjami z dodatniego przedziału określonego przez użytkownika. 
#include <iostream>
using namespace std;

int n, m;

int main() 
{
    cout << "Podaj wartosc minimalna : ";
    cin >> n;
    cout << "Podaj wartosc maksymalna : ";
    cin >> m;

    for (n>0; n<=m; n++)
        cout << n << " ";
    
    return 0;
}
C++ - ćwiczenia z wykorzystaniem pętli | 3 |
Napisz program, który wyświetla kwadraty kolejnych liczb naturalnych, począwszy od zera a skończywszy na kwadracie liczby podanej przez użytkownika. 
#include <iostream>
using namespace std;

int n, m=0;

int main() 
{
    cout << "Podaj liczbe : ";
    cin >> n;
        
    for (m=0; m<=n; m++)
        cout << m*m << endl;

    return 0;
}

Funkcje – proste ćwiczenia | 1 |
Napisz program, który:

> Deklaruje i inicjuje zmienne: staz = 1 oraz zarobek = 1000,
> Pyta się o przewidywany okres pracy w latach i zapisuje go do zmiennej lata,
> Wywołuje funkcje drukuj() i awans() określoną w zmiennej lata ilość razy. 
#include <iostream>
using namespace std;

int staz=1, zarobek=1000, lata;

void awans()
{
    zarobek*=1.1;
    lata--;
    staz++;
}

void drukuj()
{
    cout << "Zarabialbys " << zarobek << " po " << staz << " latach stazu." << endl;
}

int main() 
{
    
    cout << "Podaj liczbe lat : ";
    cin >> lata;
    
    while (lata>0)
    {
    drukuj();
    awans();
    }

    return 0;
}
Tablice – część 1 | 1 |
Napisz program, który pobiera od użytkownika dwie 5-cyfrowe serie liczb całkowitych i zapisuje je w dwóch tablicach. Następnie program powinien obliczyć sumę elementów w każdej tablicy i poinformować, która jest większa. 
#include <iostream>
using namespace std;

int tab[5], tab2[5], suma1, suma2, y;

void oblicz()
{
    int i=0;
    while(i<5)
    {
        suma1+=tab[i];
        i++;
        cout << suma1 << endl;
    }
}

void oblicz2()
{
    int i=0;
    while(i<5)
    {
        suma2+=tab2[i];
        i++;
        cout << suma2 << endl;
    }
}

int main() 
{
    cout << "Podaj piec liczb :" << endl;
    for(int x=0; x<5; x++)
    {
    cin >> y;
    tab[x] = y;
    }
    
    cout << "Podaj piec liczb :" << endl;
    for(int x=0; x<5; x++)
    {
    cin >> y;
    tab2[x] = y;
    }

    oblicz();
    oblicz2();

    if(suma1==suma2)
    {
        cout << "Sumy tabel sa rowne.";
        return 0;
    }

    if(suma1>suma2)
    {
        cout << "Suma pierwszej tabeli jest wieksza.";
    }
    else
    {
        cout << "Suma drugiej tabeli jest wieksza.";
    }
    
    return 0;
}

#include <iostream>
#include <cstdlib>
#include <ctime>

using namespace std;

{
    srand(time(NULL));

    cout << "Losowa liczba : " << rand() << endl;
