function funkcjaprompt() {
var color = prompt("Jaki kolor tła chcesz ?\n(czerwony,zielony,niebieski lub szary)");
    if (color == "czerwony")
        document.body.style.backgroundColor = "#b30000";
    else if (color == "zielony")
        document.body.style.backgroundColor = "#009933";
    else if (color == "niebieski")
        document.body.style.backgroundColor = "#0066ff";
    else if (color == "szary")
        document.body.style.backgroundColor = "#161616";
}
